build_up_county = {
	scope = province
	saved_scope = {
		player
	}

	is_shown = {
		AND = {
			OR = {
				has_holding_type = castle_holding
				has_holding_type = city_holding
				has_holding_type = church_holding
			}
			OR = {
				province_owner = scope:player
				province_owner.liege = scope:player
				province_owner.liege.liege = scope:player
				province_owner.liege.liege.liege = scope:player
				province_owner.top_liege = scope:player
			}
		}
	}
	is_valid = {
		always = yes
	}
	ai_is_valid = {
		always = no
	}
	effect = {
		#custom_tooltip = "builds up barony to it's maximum level based on technologies"
		build_up_county_effect = yes
	}
}

invite_local_man = {
	scope = province
	saved_scope = {
		player
	}
	is_shown = {
		OR = {
			province_owner = scope:player
			province_owner.liege = scope:player
			province_owner.liege.liege = scope:player
			province_owner.liege.liege.liege = scope:player
			province_owner.top_liege = scope:player
			AND = {
				scope:player = {
					is_landless_adventurer = yes
				}
				scope:player.location = THIS
			}
		}
	}
	is_valid = {
		always = yes
	}
	ai_is_valid = {
		always = no
	}
	effect = {
		if = {
			limit = {
				scope:player = {
					has_variable = invite_local_dynasty_var
					has_variable = invite_local_religion_var
				}
			}

			create_character = {
				age = 18
				gender = male
				employer = scope:player
				faith = scope:player.faith
				culture = root.county.culture
				dynasty = generate
				prowess = 10
				trait = physique_good_3
			}
		}
		else_if = {
			limit = {
				scope:player = {
					has_variable = invite_local_dynasty_var
				}
			}

			create_character = {
				age = 18
				gender = male
				employer = scope:player
				faith = root.county.faith
				culture = root.county.culture
				dynasty = generate
				prowess = 10
				trait = physique_good_3
			}
		}
		else_if = {
			limit = {
				scope:player = {
					has_variable = invite_local_religion_var
				}
			}

			create_character = {
				age = 18
				gender = male
				employer = scope:player
				faith = scope:player.faith
				culture = root.county.culture
				dynasty = none
				prowess = 10
				trait = physique_good_3
			}
		}
		else = {
			create_character = {
				age = 18
				gender = male
				employer = scope:player
				faith = root.county.faith
				culture = root.county.culture
				dynasty = none
				prowess = 10
				trait = physique_good_3
			}
		}
	}
}
